Iterative approach :
#include <iostream>
using namespace std;


int main(){
    int n;
    cout<<"Enter number of terms "<<endl;
    cin>>n;
    
    int first = 0,second = 1,next;
    
    for(int i = 0;i<n;i++){
        if(i<=1){
            next = i;
        }
        else{
            next = first + second;
            first = second;
            second = next;
        }
        cout<<next<<" ";
    }
    return 0;
}

Recursive approach:

#include <iostream>
using namespace std;

int fibo(int n){
    if(n == 0) return 0;
    else if(n == 1) return 1;
    else{
        return fibo(n-1) + fibo(n-2);
    }
}

int main(){
    int n;
    cout<<"Enter number of terms: "<<endl;
    cin>>n;
    
    for(int i = 0;i<n;i++){
        cout<<fibo(i)<<" ";
    }
    return 0;
}
