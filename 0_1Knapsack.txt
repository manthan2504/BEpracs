#include <iostream>
#include <algorithm>
using namespace std;

int knapsack(int weight[], int profit[], int n, int capacity) {
    int dp[n+1][capacity+1];

    // Step 1: Build the DP table
    for (int i = 0; i <= n; i++) {
        for (int w = 0; w <= capacity; w++) {
            if (i == 0 || w == 0)
                dp[i][w] = 0;
            else if (weight[i-1] <= w)
                dp[i][w] = max(profit[i-1] + dp[i-1][w - weight[i-1]],
                               dp[i-1][w]);
            else
                dp[i][w] = dp[i-1][w];
        }
    }

    // Step 2: Print included items
    int res = dp[n][capacity];
    int w = capacity;
    cout << "\nItems included in knapsack:\n";
    for (int i = n; i > 0 && res > 0; i--) {
        if (res != dp[i-1][w]) {
            cout << "Item: " << i << " included Profit: " << profit[i-1]
                 << ", Weight: " << weight[i-1] << ")\n";
            res -= profit[i-1];
            w -= weight[i-1];
        }
    }

    // Step 3: Return max profit
    return dp[n][capacity];
}

int main() {
    int n, capacity;

    cout << "Enter number of items: ";
    cin >> n;

    int profit[n], weight[n];
    for (int i = 0; i < n; i++) {
        cout << "Enter profit and weight for item " << i + 1 << ": ";
        cin >> profit[i] >> weight[i];
    }

    cout << "Enter knapsack capacity: ";
    cin >> capacity;

    int maxProfit = knapsack(weight, profit, n, capacity);

    cout << "\nMaximum Profit: " << maxProfit << endl;

    return 0;
}
